cmake_minimum_required(VERSION 3.10)

set(DEVICE_NAME "ascend")
project(modelbox-devices-${DEVICE_NAME})

file(GLOB_RECURSE LIBMODELBOX_DEVICE_SOURCES *.cpp *.cc *.c)
set(LIBMODELBOX_DEVICE_ASCEND_INCLUDE ${CMAKE_CURRENT_LIST_DIR}/include)

include_directories(${CMAKE_CURRENT_LIST_DIR})
include_directories(${CMAKE_CURRENT_BINARY_DIR})
include_directories(${LIBMODELBOX_INCLUDE})
include_directories(${LIBMODELBOX_BASE_INCLUDE})
include_directories(${LIBMODELBOX_DEVICE_ASCEND_INCLUDE})
include_directories(${ACL_INCLUDE_DIR})
include_directories(${DSMI_INCLUDE_DIR})
include_directories(${MODELBOX_COMMON_DEVICE_STREAM_INCLUDE})

set(HEADER 
    ${LIBMODELBOX_DEVICE_ASCEND_INCLUDE}/modelbox
)

set(LIBMODELBOX_DEVICE_ASCEND_STATIC libmodelbox-device-${DEVICE_NAME}-static)
set(LIBMODELBOX_DEVICE_ASCEND_SHARED libmodelbox-device-${DEVICE_NAME}-shared)

add_library(${LIBMODELBOX_DEVICE_ASCEND_STATIC} STATIC ${LIBMODELBOX_DEVICE_SOURCES})
add_library(${LIBMODELBOX_DEVICE_ASCEND_SHARED} SHARED ${LIBMODELBOX_DEVICE_SOURCES})

set_target_properties(${LIBMODELBOX_DEVICE_ASCEND_SHARED} PROPERTIES 
    SOVERSION ${MODELBOX_VERSION_MAJOR}
    VERSION ${MODELBOX_VERSION_MAJOR}.${MODELBOX_VERSION_MINOR}.${MODELBOX_VERSION_PATCH}
)

target_link_libraries(${LIBMODELBOX_DEVICE_ASCEND_STATIC} ${LIBMODELBOX_SHARED})
target_link_libraries(${LIBMODELBOX_DEVICE_ASCEND_STATIC} pthread)
target_link_libraries(${LIBMODELBOX_DEVICE_ASCEND_STATIC} rt)
target_link_libraries(${LIBMODELBOX_DEVICE_ASCEND_STATIC} dl)
target_link_libraries(${LIBMODELBOX_DEVICE_ASCEND_STATIC} ${ACL_LIBRARIES})
target_link_libraries(${LIBMODELBOX_DEVICE_ASCEND_STATIC} ${DSMI_LIBRARIES})
target_link_libraries(${LIBMODELBOX_DEVICE_ASCEND_STATIC} ${MODELBOX_COMMON_DEVICE_STREAM_LIBRARY})

target_link_libraries(${LIBMODELBOX_DEVICE_ASCEND_SHARED} ${LIBMODELBOX_SHARED})
target_link_libraries(${LIBMODELBOX_DEVICE_ASCEND_SHARED} pthread)
target_link_libraries(${LIBMODELBOX_DEVICE_ASCEND_SHARED} rt)
target_link_libraries(${LIBMODELBOX_DEVICE_ASCEND_SHARED} dl)
target_link_libraries(${LIBMODELBOX_DEVICE_ASCEND_SHARED} ${ACL_LIBRARIES})
target_link_libraries(${LIBMODELBOX_DEVICE_ASCEND_SHARED} ${DSMI_LIBRARIES})
target_link_libraries(${LIBMODELBOX_DEVICE_ASCEND_SHARED} ${MODELBOX_COMMON_DEVICE_STREAM_LIBRARY})

set_target_properties(${LIBMODELBOX_DEVICE_ASCEND_STATIC} ${LIBMODELBOX_DEVICE_ASCEND_SHARED} 
    PROPERTIES OUTPUT_NAME "modelbox-device-${DEVICE_NAME}"
)
set_target_properties(${LIBMODELBOX_DEVICE_ASCEND_STATIC} ${LIBMODELBOX_DEVICE_ASCEND_SHARED}
    PROPERTIES
    ARCHIVE_OUTPUT_DIRECTORY "${TEST_WORKING_LIB_DIR}"
    RUNTIME_OUTPUT_DIRECTORY "${TEST_WORKING_BIN_DIR}"
)

configure_file(${CMAKE_CURRENT_LIST_DIR}/libmodelbox-device-${DEVICE_NAME}.pc.in ${CMAKE_CURRENT_BINARY_DIR}/libmodelbox-device-${DEVICE_NAME}.pc @ONLY)

install(TARGETS ${LIBMODELBOX_DEVICE_ASCEND_SHARED} 
    COMPONENT ascend-device-flowunit
    RUNTIME DESTINATION ${CMAKE_INSTALL_FULL_BINDIR}
    LIBRARY DESTINATION ${CMAKE_INSTALL_FULL_LIBDIR}
    OPTIONAL)

install(TARGETS ${LIBMODELBOX_DEVICE_ASCEND_STATIC} 
    COMPONENT ascend-device-flowunit-devel
    ARCHIVE DESTINATION ${CMAKE_INSTALL_FULL_LIBDIR}
    OPTIONAL)

install(DIRECTORY 
    ${HEADER} 
    DESTINATION ${CMAKE_INSTALL_FULL_INCLUDEDIR} 
    COMPONENT ascend-device-flowunit-devel
    )

install(FILES 
    ${CMAKE_CURRENT_BINARY_DIR}/libmodelbox-device-${DEVICE_NAME}.pc 
    DESTINATION ${CMAKE_INSTALL_LIBDIR}/pkgconfig 
    COMPONENT ascend-device-flowunit-devel
    )

set(LIBMODELBOX_DEVICE_ASCEND_STATIC ${LIBMODELBOX_DEVICE_ASCEND_STATIC} CACHE INTERNAL "")
set(LIBMODELBOX_DEVICE_ASCEND_SHARED ${LIBMODELBOX_DEVICE_ASCEND_SHARED} CACHE INTERNAL "")
set(LIBMODELBOX_DEVICE_ASCEND_INCLUDE ${LIBMODELBOX_DEVICE_ASCEND_INCLUDE} CACHE INTERNAL "")
set(LIBMODELBOX_DEVICE_ASCEND_BINARY_DIR ${CMAKE_CURRENT_BINARY_DIR} CACHE INTERNAL "")
set(LIBMODELBOX_DEVICE_SOURCES ${LIBMODELBOX_DEVICE_SOURCES} CACHE INTERNAL "")
set(LIBMODELBOX_DEVICE_ASCEND_SO_PATH ${CMAKE_CURRENT_BINARY_DIR}/libmodelbox-device-${DEVICE_NAME}.so CACHE INTERNAL "")

# driver test
list(APPEND DRIVER_UNIT_TEST_SOURCE ${MODELBOX_UNIT_TEST_SOURCE})
list(APPEND DRIVER_UNIT_TEST_TARGET ${LIBMODELBOX_DEVICE_ASCEND_SHARED})
set(DRIVER_UNIT_TEST_SOURCE ${DRIVER_UNIT_TEST_SOURCE} CACHE INTERNAL "")
set(DRIVER_UNIT_TEST_TARGET ${DRIVER_UNIT_TEST_TARGET} CACHE INTERNAL "")
